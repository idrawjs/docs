## 安装模块

```sh
npm install idraw
```

## 渲染DOM的位置

```html
<div id="app"></div>
```

## 使用 iDraw.js

```js
import { iDraw } from 'idraw';
const data = {
  elements: [
    {
      name: 'rect-001',
      x: 160,
      y: 120,
      w: 200,
      h: 100,
      angle: 30,
      type: 'rect',
      detail: {
        background: '#d5f5f9',
        borderRadius: 10,
        borderWidth: 2,
        borderColor: '#3f51b5'
      }
    }
  ]
};

const app = document.querySelector('#app');
const idraw = new iDraw(app, {
  width: 600,
  height: 400,
  devicePixelRatio: 2
});

// 设置数据
idraw.setData(data);
```
